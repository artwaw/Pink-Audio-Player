429 /*!
430  * \brief Routine resposible for calculating the next track to be played.
431  *
432  * The routine takes into the account the playback mode setting.
433  * \param prev Jump direction, TRUE if jump backward. Only applicable if playback setting != random and playback doesn't follow the cursor.
434  * \return Model row of the next track or -1.
435  */
436 int Player::nextTrack(const bool prev) const {
437  if (prev) { return prevTrack(); }
438  if (!currentlyPlaying.first) { return -1; }
439  if (playFollowsCursor->isChecked()) {
440  if (!currentlyPlaying.first->selectionModel()->hasSelection()) {
441  return -1;
442  }
443  return currentlyPlaying.first->selectionModel()->selectedIndexes().at(0).row();
444  }
445  if (ui->orderBox->currentIndex()==2) {
446  return -1;
447  }
448  if (ui->orderBox->currentIndex()==0) {
449  if (currentlyPlaying.second==currentlyPlaying.first->model()->rowCount()-1) {
450  return -1;
451  }
452  return currentlyPlaying.second+1;
453  }
454  if (ui->orderBox->currentIndex()==1) {
455  if (currentlyPlaying.second==currentlyPlaying.first->model()->rowCount()-1) {
456  return 0;
457  }
458  return currentlyPlaying.second+1;
459  }
460  if (ui->orderBox->currentIndex()==3) {
461  return currentlyPlaying.second;
462  }
463  if (ui->orderBox->currentIndex()==4) {
464  return QRandomGenerator::global()->bounded(currentlyPlaying.first->model()->rowCount());
465  }
466  return -1;
467 }
